func searchInsert(nums []int, target int) int {
    n := len(nums)
    left, right := 0, n - 1
    ans := n
    for left <= right {
        mid := (right - left) >> 1 + left
        if target <= nums[mid] {
            ans = mid
            right = mid - 1
        } else {
            left = mid + 1
        }
    }
    return ans
}

func searchMatrix(matrix [][]int, target int) bool {
    var l, r , mid int
    for _, nums := range matrix {
        l = 0
        n := len(nums)
        r = n-1
        mid = n
        for l<=r {
            mid = l + (r-l)/2 
            if nums[mid] < target{
                l = mid + 1
            }
            if nums[mid]>target{
                r = mid - 1
            }
            if nums[mid]==target{
                return true
            }
        }
    }
    return false
}
