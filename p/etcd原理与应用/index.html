import (

//ServiceRegister åˆ›å»ºç§Ÿçº¦æ³¨å†ŒæœåŠ¡
type ServiceRegister struct {
	cli     *clientv3.Client //etcd client
	leaseID clientv3.LeaseID //ç§Ÿçº¦ID
	//ç§Ÿçº¦keepalieveç›¸åº”chan
	keepAliveChan <-chan *clientv3.LeaseKeepAliveResponse
	key           string //key
	val           string //value
}

//NewServiceRegister æ–°å»ºæ³¨å†ŒæœåŠ¡
func NewServiceRegister(endpoints []string, key, val string, lease int64) (*ServiceRegister, error) {
	cli, err := clientv3.New(clientv3.Config{
		Endpoints:   endpoints,
		DialTimeout: 5 * time.Second,
	})
	if err != nil {
		log.Fatal(err)
	}

	ser := &ServiceRegister{
		cli: cli,
		key: key,
		val: val,
	}

	//ç”³è¯·ç§Ÿçº¦è®¾ç½®æ—¶é—´keepalive
	if err := ser.putKeyWithLease(lease); err != nil {
		return nil, err
	}

	return ser, nil
}

//è®¾ç½®ç§Ÿçº¦
func (s *ServiceRegister) putKeyWithLease(lease int64) error {
	//è®¾ç½®ç§Ÿçº¦æ—¶é—´
	resp, err := s.cli.Grant(context.Background(), lease)
	if err != nil {
		return err
	}
	//æ³¨å†ŒæœåŠ¡å¹¶ç»‘å®šç§Ÿçº¦
	_, err = s.cli.Put(context.Background(), s.key, s.val, clientv3.WithLease(resp.ID))
	if err != nil {
		return err
	}
	//è®¾ç½®ç»­ç§Ÿ å®šæœŸå‘é€éœ€æ±‚è¯·æ±‚
	leaseRespChan, err := s.cli.KeepAlive(context.Background(), resp.ID)

	if err != nil {
		return err
	}
	s.leaseID = resp.ID
	log.Println(s.leaseID)
	s.keepAliveChan = leaseRespChan
	log.Printf("Put key:%s  val:%s  success!", s.key, s.val)
	return nil
}

//ListenLeaseRespChan ç›‘å¬ ç»­ç§Ÿæƒ…å†µ
func (s *ServiceRegister) ListenLeaseRespChan() {
	for leaseKeepResp := range s.keepAliveChan {
		log.Println("ç»­çº¦æˆåŠŸ", leaseKeepResp)
	}
	log.Println("å…³é—­ç»­ç§Ÿ")
}

// Close æ³¨é”€æœåŠ¡
func (s *ServiceRegister) Close() error {
	//æ’¤é”€ç§Ÿçº¦
	if _, err := s.cli.Revoke(context.Background(), s.leaseID); err != nil {
		return err
	}
	log.Println("æ’¤é”€ç§Ÿçº¦")
	return s.cli.Close()
}

package main

import (
	"context"
	"log"
	"sync"
	"time"

	"github.com/coreos/etcd/mvcc/mvccpb"
	"go.etcd.io/etcd/clientv3"
)

//ServiceDiscovery æœåŠ¡å‘ç°
type ServiceDiscovery struct {
	cli        *clientv3.Client  //etcd client
	serverList map[string]string //æœåŠ¡åˆ—è¡¨
	lock       sync.Mutex
}

//NewServiceDiscovery  æ–°å»ºå‘ç°æœåŠ¡
func NewServiceDiscovery(endpoints []string) *ServiceDiscovery {
	cli, err := clientv3.New(clientv3.Config{
		Endpoints:   endpoints,
		DialTimeout: 5 * time.Second,
	})
	if err != nil {
		log.Fatal(err)
	}

	return &ServiceDiscovery{
		cli:        cli,
		serverList: make(map[string]string),
	}
}

//WatchService åˆå§‹åŒ–æœåŠ¡åˆ—è¡¨å’Œç›‘è§†
func (s *ServiceDiscovery) WatchService(prefix string) error {
	//æ ¹æ®å‰ç¼€è·å–ç°æœ‰çš„key
	resp, err := s.cli.Get(context.Background(), prefix, clientv3.WithPrefix())
	if err != nil {
		return err
	}

	for _, ev := range resp.Kvs {
		s.SetServiceList(string(ev.Key), string(ev.Value))
	}

	//ç›‘è§†å‰ç¼€ï¼Œä¿®æ”¹å˜æ›´çš„server
	go s.watcher(prefix)
	return nil
}

//watcher ç›‘å¬å‰ç¼€
func (s *ServiceDiscovery) watcher(prefix string) {
	rch := s.cli.Watch(context.Background(), prefix, clientv3.WithPrefix())
	log.Printf("watching prefix:%s now...", prefix)
	for wresp := range rch {
		for _, ev := range wresp.Events {
			switch ev.Type {
			case mvccpb.PUT: //ä¿®æ”¹æˆ–è€…æ–°å¢
				s.SetServiceList(string(ev.Kv.Key), string(ev.Kv.Value))
			case mvccpb.DELETE: //åˆ é™¤
				s.DelServiceList(string(ev.Kv.Key))
			}
		}
	}
}

//SetServiceList æ–°å¢æœåŠ¡åœ°å€
func (s *ServiceDiscovery) SetServiceList(key, val string) {
	s.lock.Lock()
	defer s.lock.Unlock()
	s.serverList[key] = string(val)
	log.Println("put key :", key, "val:", val)
}

//DelServiceList åˆ é™¤æœåŠ¡åœ°å€
func (s *ServiceDiscovery) DelServiceList(key string) {
	s.lock.Lock()
	defer s.lock.Unlock()
	delete(s.serverList, key)
	log.Println("del key:", key)
}

//GetServices è·å–æœåŠ¡åœ°å€
func (s *ServiceDiscovery) GetServices() []string {
	s.lock.Lock()
	defer s.lock.Unlock()
	addrs := make([]string, 0)

	for _, v := range s.serverList {
		addrs = append(addrs, v)
	}
	return addrs
}

//Close å…³é—­æœåŠ¡
func (s *ServiceDiscovery) Close() error {
	return s.cli.Close()
}

func main() {
	var endpoints = []string{"localhost:2379"}
	ser := NewServiceDiscovery(endpoints)
	defer ser.Close()
	ser.WatchService("/web/")
	ser.WatchService("/gRPC/")
	for {
		select {
		case <-time.Tick(10 * time.Second):
			log.Println(ser.GetServices())
		}
	}
}
